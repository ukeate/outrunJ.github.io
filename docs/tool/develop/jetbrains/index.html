    settings -> Editor -> File and Code Templates -> Includes -> File Header
       /**  
        *
        * @Description: ${Description}
        * @author: ShenWenqing
        * @date: Created on ${DATE} ${TIME}
        *
        */

    settings -> Inspections

    .idea/encodings.xml里删除除了UTF-8的项
